import pandas as pd

df = pd.read_csv("train.csv")

# Aperçu
print(df.head())

# Structure
print(df.info())

# Valeurs manquantes
print(df.isnull().sum())

# Statistiques
print(df.describe())

# Analyse cible
print(df["Survived"].value_counts(normalize=True))
